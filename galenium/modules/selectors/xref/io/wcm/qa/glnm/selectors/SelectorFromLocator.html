20  package io.wcm.qa.glnm.selectors;
21  
22  import com.galenframework.specs.page.Locator;
23  
24  import io.wcm.qa.glnm.selectors.base.AbstractNestedSelectorBase;
25  
26  /**
27   * Turns a Galen {@link com.galenframework.specs.page.Locator} object into a Galenium  {@link io.wcm.qa.glnm.selectors.base.Selector}.
28   *
29   * @since 1.0.0
30   */
31  public class SelectorFromLocator extends AbstractNestedSelectorBase {
32  
33    /**
34     * <p>Constructor for SelectorFromLocator.</p>
35     *
36     * @param locator to use in Selector construction
37     */
38    public SelectorFromLocator(Locator locator) {
39      this(null, locator);
40    }
41  
42    /**
43     * <p>Constructor for SelectorFromLocator.</p>
44     *
45     * @param elementName alternative name for use in reporting
46     * @param locator to use in Selector construction
47     */
48    public SelectorFromLocator(String elementName, Locator locator) {
49      setName(elementName);
50      setLocator(locator);
51    }
52  
53    @Override
54    protected void setLocator(Locator locator) {
55      super.setLocator(locator);
56      setString(locator.getLocatorValue());
57      Locator parentLocator = locator.getParent();
58      if (parentLocator != null) {
59        String parentName = getName().replaceFirst("\\.[^.]*$", "");
60        SelectorFromLocator parentSelector = new SelectorFromLocator(parentName, parentLocator);
61        setParent(parentSelector);
62        parentSelector.addChild(this);
63      }
64    }
65  
66  }
