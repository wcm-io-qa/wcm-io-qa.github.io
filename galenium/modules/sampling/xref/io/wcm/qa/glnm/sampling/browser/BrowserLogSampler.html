20  package io.wcm.qa.glnm.sampling.browser;
21  
22  import static java.util.stream.Collectors.toList;
23  
24  import java.util.Collection;
25  import java.util.logging.Level;
26  
27  import io.wcm.qa.glnm.interaction.Browser;
28  import io.wcm.qa.glnm.sampling.base.CachingBasedSampler;
29  
30  /**
31   * Samples browser log as separate strings.
32   *
33   * @since 4.0.0
34   */
35  public class BrowserLogSampler extends CachingBasedSampler<Iterable<String>> {
36  
37    private Level level;
38  
39    /**
40     * With default log level {@link java.util.logging.Level#SEVERE}.
41     */
42    public BrowserLogSampler() {
43      this(Level.SEVERE);
44    }
45  
46    /**
47     * <p>
48     * With custom level.
49     * </p>
50     *
51     * @param level will sample only entries at or above this level
52     */
53    public BrowserLogSampler(Level level) {
54      setLevel(level);
55    }
56  
57    @Override
58    protected Collection<String> freshSample() {
59      return Browser.getLog(getLevel())
60          .stream()
61          .map(entry -> entry.getMessage())
62          .collect(toList());
63    }
64  
65    /**
66     * <p>Getter for the field <code>level</code>.</p>
67     *
68     * @return a {@link java.util.logging.Level} object.
69     */
70    public Level getLevel() {
71      return level;
72    }
73  
74    /**
75     * <p>Setter for the field <code>level</code>.</p>
76     *
77     * @param level a {@link java.util.logging.Level} object.
78     */
79    public void setLevel(Level level) {
80      this.level = level;
81    }
82  
83  }
